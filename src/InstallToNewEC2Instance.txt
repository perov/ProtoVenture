
# Packages
sudo easy_install pip
sudo pip install flask requests
sudo yum -y install gcc-c++
sudo yum -y install make git python-devel

# Boost
wget http://downloads.sourceforge.net/project/boost/boost/1.52.0/boost_1_52_0.tar.bz2
tar --bzip2 -xf boost_1_52_0.tar.bz2
cd boost_1_52_0
sudo ./bootstrap.sh --with-libraries=thread,system
sudo ./b2 install
cd ..

# GSL
wget ftp://ftp.gnu.org/gnu/gsl/gsl-1.15.tar.gz
tar --gz -xf gsl-1.15.tar.gz
cd gsl-1.15
./configure
make
sudo make install
cd ..

# Boost and GSL
sudo sh -c 'echo "/usr/local/lib" >> /etc/ld.so.conf'
sudo ldconfig
